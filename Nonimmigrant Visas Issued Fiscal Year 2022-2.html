# Countries that requested the most Visas to the USA.
sum_nationality = dados.groupby('Nationality').sum()

# Calculate the total of all visa types by nationality
sum_nationality['Total_Vistos'] = sum_nationality.sum(axis=1)

# Sort in descending order.
sum_nationality_sort = sum_nationality.sort_values(by='Total_Vistos', ascending=False)

# View the countries that requested the most visas to the US
print("View the countries that requested the most visas to the US:")
print(sum_nationality_sort['Total_Vistos'].head(10))  # Show the first 10.

# Generating a bar chart
